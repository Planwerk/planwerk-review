package main

import (
	"encoding/json"
	"os"
)

// Config is the parsed application configuration.
type Config struct {
	Port int `json:"port"`
}

// Load reads and parses the config file at path, returning any read or parse
// error to the caller.
func Load(path string) (*Config, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}
	var cfg Config
	if err := json.Unmarshal(data, &cfg); err != nil {
		return nil, err
	}
	return &cfg, nil
}
